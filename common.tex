% Usage: \jpg[width]{path}
% Not floating
\newcommand*{\jpg}[2][0.618]{
    \begin{center}
      \includegraphics[width={#1}\textwidth]{contents/images/{#2}.jpg}
    \end{center}
}
% Floating
\newcommand*{\jpgf}[2][0.618]{
  \begin{figure}[ht!]
    \begin{center}
      \includegraphics[width={#1}\textwidth]{contents/images/{#2}.jpg}
    \end{center}
  \end{figure} 
}
% By height
\newcommand*{\jpgh}[2][0.618]{
    \begin{center}
      \includegraphics[height={#1}\textheight]{contents/images/{#2}.jpg}
    \end{center}
}

% Includ PDF file
\newcommand*{\pdf}[1]{

}
